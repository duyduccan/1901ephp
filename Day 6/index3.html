câu lệnh WHERE
    được sử dụng để filter (lọc) dữ liệu từ bảng
    ví dụ:
    SELECT * FROM customers WHERE id=5;
    (khi WHERE thì nếu cột là số thì không cần dấu nháy)
    ví dụ 2:
    SELECT * FROM customers WHERE company="Company A";
    lấy tất cả các bản ghi mà cột company="Company A"
    (nếu WHERE cột là chuỗi thì chuỗi phải để trong dấu ''hay"")
